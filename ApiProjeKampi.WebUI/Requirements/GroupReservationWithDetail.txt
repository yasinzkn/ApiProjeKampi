<script>
    document.addEventListener('DOMContentLoaded', function() {
        const detailButtons = document.querySelectorAll('.detail-btn');
        const modalElement = document.getElementById('reservationDetailModal');

        detailButtons.forEach(button => {
            button.addEventListener('click', function(e) {
                e.preventDefault();
                e.stopPropagation();

                // Get data from button attributes
                const title = this.getAttribute('data-title');
                const responsible = this.getAttribute('data-responsible');
                const email = this.getAttribute('data-email');
                const status = this.getAttribute('data-status');
                const personCount = this.getAttribute('data-person-count');
                const reservationDate = this.getAttribute('data-reservation-date');
                const lastProcess = this.getAttribute('data-last-process');
                const priority = this.getAttribute('data-priority');
                const details = this.getAttribute('data-details');
                const statusClass = this.getAttribute('data-status-class');
                const statusIcon = this.getAttribute('data-status-icon');
                const priorityIcon = this.getAttribute('data-priority-icon');

                // Update modal content
                document.getElementById('modal-title').textContent = title;
                document.querySelector('#modal-responsible span').textContent = responsible;
                document.querySelector('#modal-email span').textContent = email || 'Belirtilmemiş';
                document.querySelector('#modal-person-count span').textContent = personCount + ' Kişi';
                document.querySelector('#modal-reservation-date span').textContent = reservationDate;
                document.querySelector('#modal-last-process span').textContent = lastProcess;
                document.getElementById('modal-details').textContent = details || 'Detay bilgisi bulunmamaktadır.';

                // Update status badge
                document.getElementById('modal-status').innerHTML = `
                    <span class="badge ${statusClass} px-3 py-2 rounded-pill fs-6">
                        <i class="${statusIcon} me-1"></i>
                        ${status}
                    </span>
                `;

                // Update priority badge
                document.getElementById('modal-priority').innerHTML = `
                    <span class="badge bg-light text-dark px-3 py-2 rounded-pill fs-6">
                        <i class="${priorityIcon} me-1"></i>
                        ${priority}
                    </span>
                `;

                // Show modal using Bootstrap 5
                const modal = new bootstrap.Modal(modalElement, {
                    backdrop: true,
                    keyboard: true,
                    focus: true
                });
                modal.show();
            });
        });

        // Explicit close button handlers with direct event binding
        const closeButtons = document.querySelectorAll('[data-bs-dismiss="modal"]');
        closeButtons.forEach(btn => {
            btn.addEventListener('click', function(e) {
                e.preventDefault();
                e.stopPropagation();
                const modal = bootstrap.Modal.getInstance(modalElement);
                if (modal) {
                    modal.hide();
                }
            });
        });

        // Backdrop click handler
        modalElement.addEventListener('click', function(e) {
            if (e.target === modalElement) {
                const modal = bootstrap.Modal.getInstance(modalElement);
                if (modal) {
                    modal.hide();
                }
            }
        });
    });
</script>
